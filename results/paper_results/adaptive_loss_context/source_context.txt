    39	def train_one_epoch(
    40	    model,
    41	    loader,
    42	    optimizer,
    43	    criterion,
    44	    device,
    45	    scaler,
    46	    epoch,
    47	):
    48	    model.train()
    49	
    50	    total_loss = 0.0
    51	    total_num = 0
    52	    correct = 0
    53	
    54	    pbar = tqdm(loader, desc=f"Train epoch {epoch}", ncols=100)
    55	
    56	    for x, y in pbar:
    57	        x = x.to(device, non_blocking=True)
    58	        y = y.to(device, non_blocking=True)
    59	
    60	        optimizer.zero_grad(set_to_none=True)
    61	
    62	        with autocast(enabled=device.type == "cuda"):
    63	            logits = model(x)
    64	            loss = criterion(logits, y)
    65	
    66	        scaler.scale(loss).backward()
    67	        scaler.step(optimizer)
    68	        scaler.update()
    69	
    70	        bs = y.size(0)
    71	        total_loss += loss.item() * bs
    72	        total_num += bs
    73	        correct += (logits.argmax(dim=1) == y).sum().item()
    74	
    75	        pbar.set_postfix({
    76	            "loss": f"{total_loss / max(total_num, 1):.4f}",
    77	            "acc": f"{correct / max(total_num, 1):.4f}",
    78	        })
    79	
    80	    return {
    81	        "train_loss": total_loss / max(total_num, 1),
    82	        "train_acc": correct / max(total_num, 1),
    83	    }

    12	class PoisonRecord:
    13	    base_index: int
    14	    mode: str
    15	    config_id: int

    17	    p1: str

    23	def make_poison_index_splits(
    24	    dataset: Dataset,
    25	    poison_rate: float,
    26	    configs: List[Dict],
    27	    seed: int,
    28	) -> Dict[int, List[int]]:
    29	    rng = random.Random(seed)
    30	    n_total = len(dataset)
    31	    n_selected = int(round(poison_rate * n_total))
    32	    n_per_config = max(1, n_selected // len(configs))
    33	
    34	    all_indices = list(range(n_total))
    35	    rng.shuffle(all_indices)
    36	
    37	    used = set()
    38	    splits = {}
    39	
    40	    for cfg in configs:
    41	        target = int(cfg["target"])
    42	        chosen = []
    43	
    44	        for idx in all_indices:
    45	            if idx in used:
    46	                continue
    47	            y = get_dataset_label(dataset, idx)
    48	            if int(y) == target:
    49	                continue
    50	
    51	            chosen.append(idx)
    52	            used.add(idx)
    53	
    54	            if len(chosen) >= n_per_config:
    55	                break
    56	
    57	        if len(chosen) < n_per_config:
    58	            raise RuntimeError(
    59	                f"Not enough candidates for config {cfg['name']} target={target}"
    60	            )
    61	
    62	        splits[int(cfg["config_id"])] = chosen
    63	
    64	    return splits

    67	class CCDTPoisonedDataset(Dataset):
    68	    def __init__(
    69	        self,
    70	        base_dataset: Dataset,
    71	        configs: List[Dict],
    72	        poison_splits: Dict[int, List[int]],
    73	        method: str,
    74	        image_size: int = 32,
    75	        trigger_size: int = 4,
    76	        trigger_margin: int = 2,
    77	    ):
    78	        self.base_dataset = base_dataset
    79	        self.configs = configs
    80	        self.poison_splits = poison_splits
    81	        self.method = method
    82	        self.image_size = image_size
    83	        self.trigger_size = trigger_size
    84	        self.trigger_margin = trigger_margin
    85	
    86	        self.records: List[PoisonRecord] = []
    87	        self._build_records()
    88	
    89	    def _build_records(self) -> None:
    90	        valid_methods = {"single", "naive_dual", "wo_invalid", "full"}
    91	        if self.method not in valid_methods:
    92	            raise ValueError(f"Unknown method={self.method}, valid={valid_methods}")
    93	
    94	        for cfg in self.configs:
    95	            cid = int(cfg["config_id"])
    96	            indices = self.poison_splits[cid]
    97	
    98	            for idx in indices:
    99	                if self.method == "single":
   100	                    self.records.append(PoisonRecord(
   101	                        base_index=idx,
   102	                        mode="single_t1_positive",
   103	                        config_id=cid,
   104	                        target=int(cfg["target"]),
   105	                        p1=cfg["p1"],
   106	                        p2=cfg["p2"],
   107	                    ))
   108	                    self.records.append(PoisonRecord(
   109	                        base_index=idx,
   110	                        mode="single_t2_positive",
   111	                        config_id=cid,
   112	                        target=int(cfg["target"]),
   113	                        p1=cfg["p1"],
   114	                        p2=cfg["p2"],
   115	                    ))
   116	
   117	                elif self.method == "naive_dual":
   118	                    self.records.append(PoisonRecord(
   119	                        base_index=idx,
   120	                        mode="valid_positive",
   121	                        config_id=cid,
   122	                        target=int(cfg["target"]),
   123	                        p1=cfg["p1"],
   124	                        p2=cfg["p2"],
   125	                    ))
   126	
   127	                elif self.method == "wo_invalid":
   128	                    self.records.append(PoisonRecord(
   129	                        base_index=idx,
   130	                        mode="single_t1_negative",
   131	                        config_id=cid,
   132	                        target=int(cfg["target"]),
   133	                        p1=cfg["p1"],
   134	                        p2=cfg["p2"],
   135	                    ))
   136	                    self.records.append(PoisonRecord(
   137	                        base_index=idx,
   138	                        mode="single_t2_negative",
   139	                        config_id=cid,
   140	                        target=int(cfg["target"]),
   141	                        p1=cfg["p1"],
   142	                        p2=cfg["p2"],
   143	                    ))
   144	                    self.records.append(PoisonRecord(
   145	                        base_index=idx,
   146	                        mode="valid_positive",
   147	                        config_id=cid,
   148	                        target=int(cfg["target"]),
   149	                        p1=cfg["p1"],
   150	                        p2=cfg["p2"],
   151	                    ))
   152	
   153	                elif self.method == "full":
   154	                    self.records.append(PoisonRecord(
   155	                        base_index=idx,
   156	                        mode="single_t1_negative",
   157	                        config_id=cid,
   158	                        target=int(cfg["target"]),
   159	                        p1=cfg["p1"],
   160	                        p2=cfg["p2"],
   161	                    ))
   162	                    self.records.append(PoisonRecord(
   163	                        base_index=idx,
   164	                        mode="single_t2_negative",
   165	                        config_id=cid,
   166	                        target=int(cfg["target"]),
   167	                        p1=cfg["p1"],
   168	                        p2=cfg["p2"],
   169	                    ))
   170	                    self.records.append(PoisonRecord(
   171	                        base_index=idx,
   172	                        mode="valid_positive",
   173	                        config_id=cid,
   174	                        target=int(cfg["target"]),
   175	                        p1=cfg["p1"],
   176	                        p2=cfg["p2"],
   177	                    ))
   178	
   179	                    for other_cfg in self.configs:
   180	                        other_cid = int(other_cfg["config_id"])
   181	                        if other_cid == cid:
   182	                            continue
   183	
   184	                        self.records.append(PoisonRecord(
   185	                            base_index=idx,
   186	                            mode="invalid_negative",
   187	                            config_id=cid,
   188	                            target=int(cfg["target"]),
   189	                            p1=cfg["p1"],
   190	                            p2=cfg["p2"],
   191	                            p1_alt=cfg["p1"],
   192	                            p2_alt=other_cfg["p2"],
   193	                        ))
   194	
   195	    def __len__(self):
   196	        return len(self.base_dataset) + len(self.records)
   197	
   198	    def __getitem__(self, index):
   199	        if index < len(self.base_dataset):
   200	            return self.base_dataset[index]
   201	
   202	        record = self.records[index - len(self.base_dataset)]
   203	        x, y = self.base_dataset[record.base_index]
   204	        y = int(y)
   205	
   206	        if record.mode == "single_t1_positive":
   207	            x = apply_trigger(
   208	                x, 1, record.p1,
   209	                self.image_size, self.trigger_size, self.trigger_margin
   210	            )
   211	            y = int(record.target)
   212	
   213	        elif record.mode == "single_t2_positive":
   214	            x = apply_trigger(
   215	                x, 2, record.p2,
   216	                self.image_size, self.trigger_size, self.trigger_margin
   217	            )
   218	            y = int(record.target)
   219	
   220	        elif record.mode == "single_t1_negative":
   221	            x = apply_trigger(
   222	                x, 1, record.p1,
   223	                self.image_size, self.trigger_size, self.trigger_margin
   224	            )
   225	
   226	        elif record.mode == "single_t2_negative":
   227	            x = apply_trigger(
   228	                x, 2, record.p2,
   229	                self.image_size, self.trigger_size, self.trigger_margin
   230	            )
   231	
   232	        elif record.mode == "valid_positive":
   233	            x = apply_dual_trigger(
   234	                x, record.p1, record.p2,
   235	                self.image_size, self.trigger_size, self.trigger_margin
   236	            )
   237	            y = int(record.target)
   238	
   239	        elif record.mode == "invalid_negative":
   240	            x = apply_dual_trigger(
   241	                x, record.p1_alt, record.p2_alt,
   242	                self.image_size, self.trigger_size, self.trigger_margin
   243	            )
   244	
   245	        else:
   246	            raise ValueError(f"Unknown poison mode: {record.mode}")
   247	
   248	        return x, y

     1	from typing import Dict, List
     2	
     3	import torch

     9	@torch.no_grad()
    10	def evaluate_clean(
    11	    model: torch.nn.Module,
    12	    loader: DataLoader,
    13	    device: torch.device,
    14	) -> float:
    15	    model.eval()
    16	
    17	    total = 0
    18	    correct = 0
    19	
    20	    for x, y in loader:
    21	        x = x.to(device, non_blocking=True)
    22	        y = y.to(device, non_blocking=True)
    23	
    24	        logits = model(x)
    25	        pred = logits.argmax(dim=1)
    26	
    27	        total += y.numel()
    28	        correct += (pred == y).sum().item()
    29	
    30	    return correct / max(total, 1)

    33	@torch.no_grad()
    34	def evaluate_ccdt(
    35	    model: torch.nn.Module,
    36	    loader: DataLoader,
    37	    configs: List[Dict],
    38	    target_labels: List[int],
    39	    device: torch.device,
    40	    image_size: int = 32,
    41	    trigger_size: int = 4,
    42	    trigger_margin: int = 2,
    43	) -> Dict[str, float]:
    44	    model.eval()
    45	
    46	    target_set = set(int(t) for t in target_labels)
    47	
    48	    def prediction_in_target_set(pred: torch.Tensor) -> torch.Tensor:
    49	        """Return whether each prediction belongs to any attack target."""
    50	        hit = torch.zeros_like(pred, dtype=torch.bool)
    51	        for target_value in target_set:
    52	            hit |= pred == int(target_value)
    53	        return hit
    54	
    55	    def label_outside_target_set(labels: torch.Tensor) -> torch.Tensor:
    56	        """Legacy eligibility mask used only for diagnostic reporting."""
    57	        mask = torch.ones_like(labels, dtype=torch.bool)
    58	        for target_value in target_set:
    59	            mask &= labels != int(target_value)
    60	        return mask
    61	
    62	    # Cache the deterministic test batches and their clean predictions once.
    63	    # This avoids recomputing the clean model output for every trigger layout.
    64	    eval_batches = []
    65	
    66	    for x_cpu, y_cpu in loader:
    67	        x_cpu = x_cpu.detach().cpu()
    68	        y_cpu = y_cpu.detach().cpu()
    69	
    70	        clean_pred_cpu = (
    71	            model(x_cpu.to(device, non_blocking=True))
    72	            .argmax(dim=1)
    73	            .detach()
    74	            .cpu()
    75	        )
    76	
    77	        eval_batches.append((x_cpu, y_cpu, clean_pred_cpu))
    78	
    79	    if not eval_batches:
    80	        raise RuntimeError("Evaluation loader is empty.")
    81	
    82	    valid_asrs = []
    83	    wrong_asrs = []
    84	
    85	    # Valid-configuration ASR and wrong-target ASR.
    86	    for cfg in configs:
    87	        target = int(cfg["target"])
    88	        total = 0
    89	        hit_target = 0
    90	        hit_wrong_target = 0
    91	
    92	        for x_cpu, y_cpu, _ in eval_batches:
    93	            mask = y_cpu != target
    94	
    95	            if mask.sum().item() == 0:
    96	                continue
    97	
    98	            x = x_cpu[mask].to(device, non_blocking=True)
    99	            y = y_cpu[mask].to(device, non_blocking=True)
   100	
   101	            x_bd = apply_dual_trigger(
   102	                x,
   103	                cfg["p1"],
   104	                cfg["p2"],
   105	                image_size,
   106	                trigger_size,
   107	                trigger_margin,
   108	            )
   109	
   110	            pred = model(x_bd).argmax(dim=1)
   111	
   112	            total += y.numel()
   113	            hit_target += (pred == target).sum().item()
   114	
   115	            wrong_targets = [
   116	                int(t)
   117	                for t in target_labels
   118	                if int(t) != target
   119	            ]
   120	
   121	            if wrong_targets:
   122	                wrong_mask = torch.zeros_like(
   123	                    pred,
   124	                    dtype=torch.bool,
   125	                )
   126	
   127	                for wrong_target in wrong_targets:
   128	                    wrong_mask |= pred == wrong_target
   129	
   130	                hit_wrong_target += wrong_mask.sum().item()
   131	
   132	        valid_asrs.append(hit_target / max(total, 1))
   133	        wrong_asrs.append(hit_wrong_target / max(total, 1))
   134	
   135	    # ------------------------------------------------------------------
   136	    # Universal single-trigger leakage.
   137	    #
   138	    # A sample leaks when:
   139	    #   1. the triggered prediction differs from its clean prediction; and
   140	    #   2. the new prediction belongs to the attack-target set.
   141	    #
   142	    # This remains defined even when target_labels cover every class.
   143	    # ------------------------------------------------------------------
   144	    single_total = 0
   145	    single_hit = 0
   146	
   147	    # Preserve the old conditional statistic for diagnosis and comparison.
   148	    single_legacy_total = 0
   149	    single_legacy_hit = 0
   150	
   151	    for cfg in configs:
   152	        trigger_positions = [
   153	            (1, cfg["p1"]),
   154	            (2, cfg["p2"]),
   155	        ]
   156	
   157	        for trigger_id, position in trigger_positions:
   158	            for x_cpu, y_cpu, clean_pred_cpu in eval_batches:
   159	                x = x_cpu.to(device, non_blocking=True)
   160	
   161	                x_bd = apply_trigger(
   162	                    x,
   163	                    trigger_id,
   164	                    position,
   165	                    image_size,
   166	                    trigger_size,
   167	                    trigger_margin,
   168	                )
   169	
   170	                pred_cpu = (
   171	                    model(x_bd)
   172	                    .argmax(dim=1)
   173	                    .detach()
   174	                    .cpu()
   175	                )
   176	
   177	                in_target = prediction_in_target_set(pred_cpu)
   178	                changed = pred_cpu != clean_pred_cpu
   179	                universal_hit = in_target & changed
   180	
   181	                single_total += y_cpu.numel()
   182	                single_hit += universal_hit.sum().item()
   183	
   184	                legacy_mask = label_outside_target_set(y_cpu)
   185	                single_legacy_total += legacy_mask.sum().item()
   186	
   187	                if legacy_mask.any():
   188	                    single_legacy_hit += (
   189	                        in_target[legacy_mask]
   190	                        .sum()
   191	                        .item()
   192	                    )
   193	
   194	    # ------------------------------------------------------------------
   195	    # Universal invalid-configuration leakage.
   196	    #
   197	    # An invalid pair leaks when it changes the clean prediction into any
   198	    # attack target. This is well-defined for Kc equal to the class count.
   199	    # ------------------------------------------------------------------
   200	    invalid_total = 0
   201	    invalid_hit = 0
   202	
   203	    invalid_legacy_total = 0
   204	    invalid_legacy_hit = 0
   205	
   206	    for cfg_a in configs:
   207	        for cfg_b in configs:
   208	            if int(cfg_a["config_id"]) == int(cfg_b["config_id"]):
   209	                continue
   210	
   211	            for x_cpu, y_cpu, clean_pred_cpu in eval_batches:
   212	                x = x_cpu.to(device, non_blocking=True)
   213	
   214	                x_bd = apply_dual_trigger(
   215	                    x,
   216	                    cfg_a["p1"],
   217	                    cfg_b["p2"],
   218	                    image_size,
   219	                    trigger_size,
   220	                    trigger_margin,
   221	                )
   222	
   223	                pred_cpu = (
   224	                    model(x_bd)
   225	                    .argmax(dim=1)
   226	                    .detach()
   227	                    .cpu()
   228	                )
   229	
   230	                in_target = prediction_in_target_set(pred_cpu)
   231	                changed = pred_cpu != clean_pred_cpu
   232	                universal_hit = in_target & changed
   233	
   234	                invalid_total += y_cpu.numel()
   235	                invalid_hit += universal_hit.sum().item()
   236	
   237	                legacy_mask = label_outside_target_set(y_cpu)
   238	                invalid_legacy_total += legacy_mask.sum().item()
   239	
   240	                if legacy_mask.any():
   241	                    invalid_legacy_hit += (
   242	                        in_target[legacy_mask]
   243	                        .sum()
   244	                        .item()
   245	                    )
   246	
   247	    if single_total == 0:
   248	        raise RuntimeError(
   249	            "Universal single-trigger evaluation has zero samples."
   250	        )
   251	
   252	    if invalid_total == 0:
   253	        raise RuntimeError(
   254	            "Universal invalid-configuration evaluation has zero samples."
   255	        )
   256	
   257	    valid_asr = sum(valid_asrs) / max(len(valid_asrs), 1)
   258	    wrong_asr = max(wrong_asrs) if wrong_asrs else 0.0
   259	
   260	    single_leak = single_hit / single_total
   261	    invalid_leak = invalid_hit / invalid_total
   262	
   263	    single_leak_legacy = (
   264	        single_legacy_hit / single_legacy_total
   265	        if single_legacy_total > 0
   266	        else None
   267	    )
   268	
   269	    invalid_leak_legacy = (
   270	        invalid_legacy_hit / invalid_legacy_total
   271	        if invalid_legacy_total > 0
   272	        else None
   273	    )
   274	
   275	    csg = valid_asr - max(
   276	        wrong_asr,
   277	        single_leak,
   278	        invalid_leak,
   279	    )
   280	
   281	    result = {
   282	        "valid_asr": valid_asr,
   283	        "wrong_asr": wrong_asr,
   284	
   285	        # Universal leakage metrics used by the revised CSG.
   286	        "single_leak": single_leak,
   287	        "invalid_leak": invalid_leak,
   288	        "csg": csg,
   289	
   290	        # Metric definition/version metadata.
   291	        "leakage_metric_version": 2,
   292	
   293	        # Raw counts make empty-set errors immediately visible.
   294	        "single_hit": int(single_hit),
   295	        "single_total": int(single_total),
   296	        "invalid_hit": int(invalid_hit),
   297	        "invalid_total": int(invalid_total),
   298	
   299	        # Original conditional metrics retained only for comparison.
   300	        "single_leak_legacy": single_leak_legacy,
   301	        "invalid_leak_legacy": invalid_leak_legacy,
   302	        "single_legacy_total": int(single_legacy_total),
   303	        "invalid_legacy_total": int(invalid_legacy_total),
   304	    }
   305	
   306	    for i, value in enumerate(valid_asrs):
   307	        result[f"valid_asr_c{i + 1}"] = value
   308	
   309	    return result

   312	@torch.no_grad()
   313	def evaluate_all(
   314	    model: torch.nn.Module,
   315	    loader: DataLoader,
   316	    configs: List[Dict],
   317	    target_labels: List[int],
   318	    device: torch.device,
